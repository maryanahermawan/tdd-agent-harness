    <tables>
      <businesses description="PostgreSQL core relational entity for geospatial discovery">
        ```typescript
        interface Business {
        id: string; // Primary Key (e.g., "cafe_123")
        address: string; // Full address for visitor
        postal_code: string;
        name: string; // Public display name of the establishment
        description: string; // Up to 280 characters, static owner description
        latitude: number; // WGS84 standard coordinate (-90 to 90)

        semantic_search_blob: string; // Density-optimized human readable summary of reviews

        }
        ```

      <description>This is public and not secure</description>
      <tasks>
        <task>Use Voyage AI (via the official 'voyageai' npm package) to generate query embeddings.</task>
        <task>Configure the spatial index 'idx_business_spatial' on the 'coords' column (using
          GiST).</task>
        <task>Configure the vector similarity index 'idx_business_hnsw_vibe' on the
          'embedding_vector' column.</task>
        <task>
          <description>When an HTTP GET search request arrives with user preference text, execute
            the following runtime execution flow:</description>
          <subtasks>
            <subtask>Request a text embedding for the raw preference query from Voyage AI
              `voyage-4-lite` model to
              generate 'user_query_embedding'.</subtask>
            <subtask>
              Execute the native pgvector nearest-neighbor search query. Note the use of the CDATA
              wrapper to protect the spatial operators and prevents parsing errors:
            <![CDATA[
SELECT 
    id, 
    name, 
    semantic_search_blob,
    -- The <=> operator calculates (1 - Cosine Similarity) natively
    (embedding_vector <=> :user_query_embedding) AS distance_score
FROM 
    business
WHERE 
    -- PostGIS 1km filter happens first
    ST_DWithin(coords, ST_MakePoint(:lon, :lat)::geography, 1000)
ORDER BY 
    distance_score ASC
LIMIT 5;
          ]]>
        </subtask>
            <subtask>Strict Parameter Order: Ensure longitude and latitude are passed to
              'ST_MakePoint' in (X, Y) sequence: ST_MakePoint(longitude, latitude).</subtask>
          </subtasks>
        </task>
      </tasks>
    </step>
